ExpenseCard.jsx (React.js)
// React Component State & Props
import React, { useState } from 'react';

const ExpenseCard = ({ title, amount }) => {
  const [isPaid, setIsPaid] = useState(false);

  return (
    <div className={`card ${isPaid ? 'paid' : 'pending'}`}>
      <h3>{title}</h3>
      <p>₹{amount}</p>
      <button onClick={() => setIsPaid(!isPaid)}>
        {isPaid ? 'Settled' : 'Mark as Paid'}
      </button>
    </div>
  );
};
export default ExpenseCard;
